import React, { useState, useEffect } from "react";
import { Link } from "react-router-dom";
import { createPageUrl } from "@/utils";
import { User } from "@/entities/User";
import { Idea } from "@/entities/Idea";
import { 
  Lightbulb, 
  Users, 
  Zap, 
  Target,
  TrendingUp,
  Heart,
  MessageCircle,
  ArrowRight,
  Star,
  Sparkles,
  Plus
} from "lucide-react";
import { Button } from "@/components/ui/button";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Badge } from "@/components/ui/badge";
import { motion } from "framer-motion";

const successStories = [
  {
    title: "EcoSmart Packaging",
    author: "Sarah Chen",
    impact: "Reduced plastic waste by 40% in local businesses",
    likes: 234,
    category: "Environmental"
  },
  {
    title: "AI Study Buddy",
    author: "Marcus Johnson", 
    impact: "Helped 10,000+ students improve grades",
    likes: 189,
    category: "Education"
  },
  {
    title: "Community Garden Network",
    author: "Elena Rodriguez",
    impact: "Created 50+ urban gardens across the city",
    likes: 156,
    category: "Community"
  }
];

      const currentUser = await User.me();

    } catch (error) {
      setUser(null);

    try {
      const ideas = await Idea.list("-likes_count", 6);
      setTopIdeas(ideas);
    } catch (error) {
      console.error("Failed to load ideas:", error);
    }
    
    setIsLoading(false);
  };

    try {
      await User.loginWithRedirect(window.location.href);
    } catch (error) {
      console.error("Login failed:", error);
    }
  };

    if (!user) {
      alert("Please sign in to view full idea details and interact with the community!");
      return;
    }
  };

      {/* Hero Section */}
      <section className="relative py-20 px-4 sm:px-6 lg:px-8 overflow-hidden">

              Share your ideas, collaborate with others, and turn vision into impact.
            </p>
            <div className="flex flex-col sm:flex-row gap-4 justify-center">
              {user ? (
                <>
                  <Link to={createPageUrl("Submit")}>
                    <Button className="bg-gradient-to-r from-blue-600 to-purple-600 hover:from-blue-700 hover:to-purple-700 text-white font-semibold px-8 py-3 text-lg">
                      <Plus className="w-5 h-5 mr-2" />
                      Submit Your Idea
                    </Button>
                  </Link>
                  <Link to={createPageUrl("Explore")}>
                    <Button variant="outline" className="border-white/30 text-white hover:bg-white/10 font-semibold px-8 py-3 text-lg">
                      Explore Ideas
                      <ArrowRight className="w-5 h-5 ml-2" />
                    </Button>
                  </Link>
                </>
              ) : (
                <>
                  <Button 
                    onClick={handleLogin}
                    className="bg-gradient-to-r from-blue-600 to-purple-600 hover:from-blue-700 hover:to-purple-700 text-white font-semibold px-8 py-3 text-lg"
                  >
                    <Sparkles className="w-5 h-5 mr-2" />
                    Join BroCoders
                  </Button>
                  <Button 
                    variant="outline" 
                    className="border-white/30 text-white hover:bg-white/10 font-semibold px-8 py-3 text-lg"
                    onClick={() => document.getElementById('about').scrollIntoView({ behavior: 'smooth' })}

                    Learn More
                    <ArrowRight className="w-5 h-5 ml-2" />
                  </Button>

            transition={{ duration: 0.8 }}

            className="text-center mb-16"

              We're not just another platform. We're a movement that believes every idea has the power to change the world.
            </p>
          </motion.div>

          <div className="grid md:grid-cols-3 gap-8">
            {[
              {
                icon: Lightbulb,

              {
                icon: Users,

                description: "Join a community of like-minded innovators. Like, comment, and build upon each other's ideas to create something extraordinary together."
              },

                      <feature.icon className="w-8 h-8 text-white" />
                    </div>
                    <CardTitle className="text-white text-xl mb-2">{feature.title}</CardTitle>
                  </CardHeader>
                  <CardContent>
                    <p className="text-gray-300 text-center leading-relaxed">{feature.description}</p>
                  </CardContent>
                </Card>
              </motion.div>

        <div className="max-w-7xl mx-auto">

            initial={{ opacity: 0, y: 30 }}
            whileInView={{ opacity: 1, y: 0 }}
            transition={{ duration: 0.8 }}
            viewport={{ once: true }}
            className="text-center mb-16"

            <h2 className="text-3xl md:text-4xl font-bold text-white mb-6">
              Success Stories

              Real ideas from our community that made a difference in the world.
            </p>
          </motion.div>

          <div className="grid md:grid-cols-3 gap-8">
            {successStories.map((story, index) => (
              <motion.div
                key={story.title}
                initial={{ opacity: 0, y: 30 }}
                whileInView={{ opacity: 1, y: 0 }}
                transition={{ duration: 0.8, delay: index * 0.2 }}
                viewport={{ once: true }}
              >
                <Card className="bg-white/10 backdrop-blur-sm border-white/20 hover:bg-white/15 transition-all duration-300 h-full group">
                  <CardHeader>
                    <div className="flex items-center justify-between mb-4">
                      <Badge className="bg-green-600/20 text-green-400 border-green-400/30">

                      <div className="h-4 bg-white/10 rounded"></div>
                      <div className="h-4 bg-white/10 rounded w-5/6"></div>
                    </div>
                  </CardContent>
                </Card>
              ))}
            </div>
          ) : (
            <div className="grid md:grid-cols-2 lg:grid-cols-3 gap-8">

                <motion.div
                  key={idea.id}
                  initial={{ opacity: 0, y: 30 }}
                  whileInView={{ opacity: 1, y: 0 }}
                  transition={{ duration: 0.8, delay: index * 0.1 }}

            <motion.div

              transition={{ duration: 0.8 }}

                </Button>
              </Link>
            </motion.div>

                Ready to Share Your Next Big Idea?
              </h2>
              <p className="text-xl text-gray-300 mb-8 max-w-2xl mx-auto">
                Join thousands of innovators who are already making a difference. 
                Your idea could be the next breakthrough the world needs.
              </p>
              <Button 
                onClick={handleLogin}
                className="bg-gradient-to-r from-blue-600 to-purple-600 hover:from-blue-700 hover:to-purple-700 text-white font-semibold px-12 py-4 text-lg"

                <Zap className="w-6 h-6 mr-2" />

              </Button>
            </motion.div>

        </section>
